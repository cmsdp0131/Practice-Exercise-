Crypto Tracker

body { font-family: Arial, sans-serif; display: flex; justify-content: center; align-items: center; height: 100vh; margin: 0; background-color: #f0f0f0; } #crypto-container { text-align: center; background-color: #fff; padding: 20px; border-radius: 10px; box-shadow: 0 0 10px rgba(0, 0, 0, 0.1); } document.addEventListener('DOMContentLoaded', function() { fetch('https://api.coingecko.com/api/v3/coins/markets?vs_currency=usd&ids=bitcoin') .then(response => response.json()) .then(data => { const cryptoData = data[0]; const cryptoContainer = document.getElementById('crypto-container'); cryptoContainer.innerHTML = `

Bitcoin (BTC)

Price: $${cryptoData.current_price}

Market Cap: $${cryptoData.market_cap}

`; }) .catch(error => console.error('Error fetching data:', error)); });
